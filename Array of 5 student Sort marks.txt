#include<stdio.h>
typedef struct{
    int roll;
    char name[10];
    float mark;
}Student; 
void sort(Student s[],int n)
{
    int i,j;
    for(i=0;i<n;i++)
    {
        for(j=0;j<n;j++)
        {
            if(s[i].mark<s[j].mark)
            {
                Student temp=s[i];
                s[i]=s[j];
                s[j]=temp;
            }
        }
    }
}
void print(Student s[],int n)
{
    for(int i=0;i<n;i++)
    {
        printf("%d %s %f\n",s[i].roll,s[i].name,s[i].mark);
    }
}
int main()
{
    Student st[5]={
        {1,"hi",93.6},
        {2,"bye",96.6},
        {3,"hello",99.5},
        {4,"and",95.5},
        {5,"dude",91.1}
    };
    sort(st,5);
    print(st,5);
}
